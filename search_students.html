    // Helper: convert ArrayBuffer to hex string

    function bufferToHex(buffer) {

      const bytes = new Uint8Array(buffer);

      const hex = Array.from(bytes, b => b.toString(16).padStart(2,'0')).join('');

      return hex;

    }


    async function sha256Hex(message) {

      // message is string

      const enc = new TextEncoder();

      const data = enc.encode(message);

      const hashBuffer = await crypto.subtle.digest('SHA-256', data);

      return bufferToHex(hashBuffer);

    }


    const resultsEl = document.getElementById('results');

    async function searchStudent(){

      const q = queryEl.value.trim();

      resultsEl.style.display = 'none';

      resultsEl.innerHTML = '';


      if(!q){

        return;

      }


      // Hash the user input (student ID)

      let inputHash;

      try {

        inputHash = await sha256Hex(q);

      } catch (err) {

        resultsEl.style.display = 'block';

        resultsEl.innerHTML = `<div class="no-match">Hashing failed: ${err}</div>`;

        return;

      }


      // Find entry with matching idHash

      const match = dataset.find(d => d.idHash === inputHash);


      if(!match){

        resultsEl.style.display = 'block';

        resultsEl.innerHTML = `<div class="no-match">No match found for the provided student ID.</div>`;

        return;

      }


      // Build a table row showing the hashed ID (for privacy), GPA, and Section

      resultsEl.style.display = 'block';

      resultsEl.innerHTML = `
